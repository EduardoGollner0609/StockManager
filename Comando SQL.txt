CREATE TABLE tb_product(
id INT PRIMARY KEY NOT NULL,
name VARCHAR(100),
description TEXT,
category VARCHAR(60),
mark VARCHAR(80),
quantity INT,
price DOUBLE,
measurement_unit VARCHAR(12)
);

CREATE TABLE tb_client(
id INT AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(100),
email VARCHAR(120),
phone VARCHAR(50),
cpf VARCHAR(20) UNIQUE
);
