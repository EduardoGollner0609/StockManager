CREATE TABLE tb_product(
id INT PRIMARY KEY NOT NULL,
name VARCHAR(100),
description TEXT,
category VARCHAR(60),
mark VARCHAR(80),
quantity INT,
price DOUBLE,
measurement_unit VARCHAR(12)
);

CREATE TABLE tb_client(
id INT AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(100),
phone VARCHAR(50)
);

CREATE TABLE tb_order(
id INT AUTO_INCREMENT PRIMARY KEY,
moment TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

ALTER TABLE tb_order ADD COLUMN client_id INT NOT NULL, 
ADD CONSTRAINT fk_order_client FOREIGN KEY(client_id) REFERENCES tb_client(id);

ALTER TABLE tb_order ADD COLUMN product_id INT NOT NULL,
ADD CONSTRAINT fk_order_product FOREIGN KEY(product_id) REFERENCES tb_product(id);
